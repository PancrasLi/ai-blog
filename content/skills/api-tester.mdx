## 介绍

1. **安装依赖**

2. **编写 API 测试**
```javascript
// 导入测试工具
const axios = require('axios');

class APITester {
  constructor(baseUrl) {
    this.baseUrl = baseUrl;
  }

  async testGETEndpoint(endpoint, params = {}) {
    try {
      const response = await axios.get(`${this.baseUrl}${endpoint}`, {
        params,
        timeout: 5000,
      });
      return {
        status: response.status,
        data: response.data,
        headers: response.headers,
      };
    } catch (error) {
      return { error: error.message };
    }
  }

  async testPOSTEndpoint(endpoint, payload) {
    const response = await axios.post(`${this.baseUrl}${endpoint}`, payload, {
      headers: { 'Content-Type': 'application/json' },
    });
    return response.data;
  }

  async testWithAuthentication(endpoint, token) {
    const response = await axios.get(`${this.baseUrl}${endpoint}`, {
      headers: { Authorization: `Bearer ${token}` },
    });
    return response.data;
  }
}

// 使用示例
const tester = new APITester('https://api.example.com');
const result = await tester.testGETEndpoint('/users', { page: 1 });
console.log(result);
```

```typescript
import axios from 'axios';

describe('API 端点测试', () => {
  const API_BASE = 'https://jsonplaceholder.typicode.com';

  test('GET /posts 返回成功', async () => {
    const response = await axios.get(`${API_BASE}/posts/1`);
    expect(response.status).toBe(200);
    expect(response.data).toHaveProperty('id');
    expect(response.data).toHaveProperty('title');
  });

  test('POST /posts 创建新记录', async () => {
    const payload = {
      title: '新文章',
      body: '测试内容',
      userId: 1,
    };
    const response = await axios.post(`${API_BASE}/posts`, payload);
    expect(response.status).toBe(201);
    expect(response.data.title).toBe(payload.title);
  });

  test('错误处理 - 404 状态码', async () => {
    try {
      await axios.get(`${API_BASE}/posts/99999`);
    } catch (error) {
      expect(error.response.status).toBe(404);
    }
  });
});
```

## 完整示例

### 示例 1：完整 API 测试套件

```javascript
const axios = require('axios');

class ComprehensiveAPITester {
  constructor(baseUrl) {
    this.baseUrl = baseUrl;
    this.results = [];
  }

  async runFullTestSuite() {
    console.log('开始 API 测试套件...');

    // 测试 1: 基本 GET 请求
    await this.testBasicGET();

    // 测试 2: 认证
    await this.testAuthentication();

    // 测试 3: 错误处理
    await this.testErrorHandling();

    // 测试 4: 性能
    await this.testPerformance();

    return this.generateReport();
  }

  async testBasicGET() {
    const start = Date.now();
    const response = await axios.get(`${this.baseUrl}/api/data`);
    const duration = Date.now() - start;

    this.results.push({
      name: 'Basic GET',
      status: response.status === 200 ? 'PASS' : 'FAIL',
      duration: `${duration}ms`,
    });
  }

  async testAuthentication() {
    try {
      const response = await axios.get(`${this.baseUrl}/api/protected`, {
        headers: { Authorization: 'Bearer invalid_token' },
      });
      this.results.push({ name: 'Auth (invalid)', status: 'FAIL' });
    } catch (error) {
      if (error.response.status === 401) {
        this.results.push({ name: 'Auth Check', status: 'PASS' });
      }
    }
  }

  async testErrorHandling() {
    try {
      await axios.get(`${this.baseUrl}/api/nonexistent`);
    } catch (error) {
      this.results.push({
        name: '404 Handling',
        status: error.response.status === 404 ? 'PASS' : 'FAIL',
      });
    }
  }

  async testPerformance() {
    const times = [];
    for (let i = 0; i < 10; i++) {
      const start = Date.now();
      await axios.get(`${this.baseUrl}/api/data`);
      times.push(Date.now() - start);
    }
    const avgTime = times.reduce((a, b) => a + b) / times.length;
    this.results.push({
      name: 'Performance',
      avgTime: `${avgTime.toFixed(2)}ms`,
      status: avgTime < 1000 ? 'PASS' : 'FAIL',
    });
  }

  generateReport() {
    console.table(this.results);
    return this.results;
  }
}

// 执行测试
const tester = new ComprehensiveAPITester('https://api.example.com');
await tester.runFullTestSuite();
```
