```javascript
// 导入视频处理库
const ffmpeg = require('fluent-ffmpeg');
const path = require('path');

class VideoClipper {
  constructor(ffmpegPath = 'ffmpeg') {
    ffmpeg.setFfmpegPath(ffmpegPath);
  }

  // 提取视频片段
  async clipVideo(inputPath, outputPath, startTime, duration) {
    return new Promise((resolve, reject) => {
      ffmpeg(inputPath)
        .setStartTime(startTime)
        .setDuration(duration)
        .output(outputPath)
        .on('end', () => {
          console.log('视频片段创建完成');
          resolve(outputPath);
        })
        .on('error', reject)
        .run();
    });
  }

  // 转换视频格式
  async convertFormat(inputPath, outputPath, format = 'mp4') {
    return new Promise((resolve, reject) => {
      ffmpeg(inputPath)
        .output(outputPath)
        .outputFormat(format)
        .on('end', () => resolve(outputPath))
        .on('error', reject)
        .run();
    });
  }

  // 提取音频
  async extractAudio(inputPath, outputPath) {
    return new Promise((resolve, reject) => {
      ffmpeg(inputPath)
        .noVideo()
        .audioCodec('libmp3lame')
        .audioBitrate('192')
        .output(outputPath)
        .on('end', () => resolve(outputPath))
        .on('error', reject)
        .run();
    });
  }
}

// 使用示例
const clipper = new VideoClipper();
await clipper.clipVideo('input.mp4', 'output.mp4', 10, 30); // 从第10秒开始，剪辑30秒
```

## 完整示例

### 示例 1：批量视频转换和优化

```javascript
const ffmpeg = require('fluent-ffmpeg');
const fs = require('fs');
const path = require('path');

class VideoBatchProcessor {
  async processBatch(inputDir, outputDir) {
    const files = fs.readdirSync(inputDir);
    const results = [];

    for (const file of files) {
      if (!file.endsWith('.mov') && !file.endsWith('.avi')) continue;

      const inputPath = path.join(inputDir, file);
      const outputPath = path.join(
        outputDir,
        file.replace(/\.[^.]*$/, '.mp4')
      );

      try {
        await this.convertToMP4(inputPath, outputPath);
        results.push({ file, status: 'SUCCESS' });
      } catch (error) {
        results.push({ file, status: 'FAILED', error: error.message });
      }
    }

    return results;
  }

  convertToMP4(inputPath, outputPath) {
    return new Promise((resolve, reject) => {
      ffmpeg(inputPath)
        .videoCodec('libx264')
        .audioCodec('aac')
        .output(outputPath)
        .on('end', resolve)
        .on('error', reject)
        .run();
    });
  }
}

// 使用
const processor = new VideoBatchProcessor();
const results = await processor.processBatch('./videos', './converted');
console.log('处理结果:', results);
```

### 示例 2：自动生成高光片段

```javascript
const ffmpeg = require('fluent-ffmpeg');

class HighlightGenerator {
  async detectAndExtractHighlights(videoPath, outputDir) {
    // 分析视频并检测高光
    const highlights = [
      { start: 120, duration: 30, title: '精彩片段 1' },
      { start: 450, duration: 45, title: '精彩片段 2' },
      { start: 890, duration: 25, title: '精彩片段 3' },
    ];

    const results = [];

    for (const highlight of highlights) {
      const outputPath = `${outputDir}/${highlight.title}.mp4`;

      await new Promise((resolve, reject) => {
        ffmpeg(videoPath)
          .setStartTime(highlight.start)
          .setDuration(highlight.duration)
          .output(outputPath)
          .on('end', () => {
            results.push({
              title: highlight.title,
              file: outputPath,
              duration: highlight.duration,
            });
            resolve();
          })
          .on('error', reject)
          .run();
      });
    }

    return results;
  }
}

// 使用
const generator = new HighlightGenerator();
const highlights = await generator.detectAndExtractHighlights(
  'long_video.mp4',
  './highlights'
);
console.log('生成的高光片段:', highlights);
```

## 常见问题

**Q: 支持哪些视频格式？**
A: 支持 MP4、WebM、MKV、MOV、AVI、FLV 等常见格式

**Q: 可以批量处理视频吗？**
A: 支持，可编写脚本批量转换、裁剪或优化

**Q: 处理大文件会很慢吗？**
A: 取决于 FFmpeg 配置，可通过调整编码器速度平衡质量和时间

**Q: 支持添加字幕吗？**
