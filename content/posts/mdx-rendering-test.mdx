interface Post {

    title: string;
    date: string;
    tags: string[];
    summary: string;
  };
}

// 获取文章函数
export async function getPostBySlug(slug: string): Promise<Post | null> {
  try {
    const posts = await getPosts();
    return posts.find((post) => post.slug === slug) || null;
  } catch (error) {
    console.error('Error:', error);
    return null;
  }
}
